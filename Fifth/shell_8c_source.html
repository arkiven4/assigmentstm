37 static char *_strtok(char *str, const char *delim, char **saveptr) {
38  char *token;
39  if (str)
40  *saveptr = str;
41  token = *saveptr;
42 
43  if (!token)
44  return NULL;
45 
46  token += strspn(token, delim);
47  *saveptr = strpbrk(token, delim);
48  if (*saveptr)
49  *(*saveptr)++ = '\0';
50 
51  return *token ? token : NULL;
52 }

136 static THD_FUNCTION(shell_thread, p) {
137  int n;
138  BaseSequentialStream *chp = ((ShellConfig *)p)->sc_channel;
139  const ShellCommand *scp = ((ShellConfig *)p)->sc_commands;
140  char *lp, *cmd, *tokp, line[SHELL_MAX_LINE_LENGTH];
141  char *args[SHELL_MAX_ARGUMENTS + 1];
142 
143  chRegSetThreadName("shell");
144 // chprintf(chp, "\r\nChibiOS/RT Shell\r\n");
145  while (true) {
146 // chprintf(chp, "ch> ");
147  if (shellGetLine(chp, line, sizeof(line))) {
148  chprintf(chp, "\r\nlogout");
149  break;
150  }
151  lp = _strtok(line, " \t", &tokp);
152  cmd = lp;
153  n = 0;
154  while ((lp = _strtok(NULL, " \t", &tokp)) != NULL) {
155  if (n >= SHELL_MAX_ARGUMENTS) {
156  chprintf(chp, "too many arguments\r\n");
157  cmd = NULL;
158  break;
159  }
160  args[n++] = lp;
161  }
162  args[n] = NULL;
163  if (cmd != NULL) {
164  if (strcasecmp(cmd, "exit") == 0) {
165  if (n > 0) {
166  usage(chp, "exit");
167  continue;
168  }
169  break;
170  }
171  else if (strcasecmp(cmd, "help") == 0) {
172  if (n > 0) {
173  usage(chp, "help");
174  continue;
175  }
176  chprintf(chp, "Commands: help exit ");
177  list_commands(chp, local_commands);
178  if (scp != NULL)
179  list_commands(chp, scp);
180  chprintf(chp, "\r\n");
181  }
182  else if (cmdexec(local_commands, chp, cmd, n, args) &&
183  ((scp == NULL) || cmdexec(scp, chp, cmd, n, args))) {
184  chprintf(chp, "%s", cmd);
185  chprintf(chp, " ?\r\n");
186  }
187  }
188  }
189  shellExit(MSG_OK);
190 }

268 bool shellGetLine(BaseSequentialStream *chp, char *line, unsigned size) {
269  char *p = line;
270 
271  while (true) {
272  char c;
273 
274  if (chSequentialStreamRead(chp, (uint8_t *)&c, 1) == 0)
275  return true;
276  if (c == 4) {
277  chprintf(chp, "^D");
278  return true;
279  }
280  if ((c == 8) || (c == 127)) {
281  if (p != line) {
282  chSequentialStreamPut(chp, c);
283  chSequentialStreamPut(chp, 0x20);
284  chSequentialStreamPut(chp, c);
285  p--;
286  }
287  continue;
288  }
289  if (c == '\r') {
290 // chprintf(chp, "\r\n");
291  *p = 0;
292  return false;
293  }
294  if (c < 0x20)
295  continue;
296  if (p < line + size - 1) {
297 // chSequentialStreamPut(chp, c);
298  *p++ = (char)c;
299  }
300  }
301 }
